计算二叉树深度

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def maxDepth(self, root):
        """
        :typeroot: TreeNode
        :rtype: int
        """
        if not root:
            return 0
        depth = 0
        level = [root]
        while level:
            depth +=1
            queue = []
            for i in level:
                if i.left:
                    queue.append(i.left)
                if i.right:
                    queue.append(i.right)
            level = queue
        return depth
